




static bool runToPosition(long p, uint16_t speed)
{
    posHandler();
    if (p > lastPosition)
    {
        
        if (CurrentPosition >= p)
        {
            stepperStop();
            return true;
        }
        else
        {
            if(isVibDone){
               isVibDone = !runToPosition(CurrentPosition-100,speed);
            }else{
               isVibDone = runToPosition(CurrentPosition+300,speed);
            }
            return false;
        }
    }
    else if (p < lastPosition)
    {
        if (CurrentPosition <= p)
        {
            stepperStop();
            return true;
        }
        else
        {
            if(isVibDone){
               isVibDone = !runToPosition(CurrentPosition+100,speed);
            }else{
               isVibDone = runToPosition(CurrentPosition-300,speed);
            }
            return false;
        }
    }
    else if (isStep)
    {
        stepperStop();
        return false;
    }
    else
    {
        return true;
    }
}
